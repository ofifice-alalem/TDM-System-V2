cd /home/motafwi1/etc/tdm.motafwiqon.com.ly && php artisan tinker --execute="
// النطاقات الأربعة
\$ranges = [
    ['CI-20260311-00048', 'CI-20260311-00065'],
    ['CI-20260311-00066', 'CI-20260311-00080'],
    ['CI-20260312-00081', 'CI-20260312-00097'],
    ['CI-20260312-00098', 'CI-20260312-00121']
];

\$allStats = [];
\$allProducts = [];

foreach (\$ranges as \$index => \$range) {
    \$invoices = App\Models\CustomerInvoice::whereBetween('invoice_number', [\$range[0], \$range[1]])
        ->where('status', 'completed')
        ->with('items.product')
        ->get();
    
    \$productTotals = [];
    foreach (\$invoices as \$invoice) {
        foreach (\$invoice->items as \$item) {
            \$productName = \$item->product->name ?? 'منتج محذوف';
            \$quantity = \$item->quantity;
            
            if (!isset(\$productTotals[\$productName])) {
                \$productTotals[\$productName] = 0;
            }
            \$productTotals[\$productName] += \$quantity;
            \$allProducts[\$productName] = true;
        }
    }
    
    \$allStats[\$index] = [
        'range' => \$range[0] . ' - ' . \$range[1],
        'invoices' => \$invoices->count(),
        'products' => \$productTotals,
        'total' => array_sum(\$productTotals)
    ];
}

// طباعة الجدول
echo str_pad('المنتج', 30) . ' | ';
for (\$i = 0; \$i < 4; \$i++) {
    echo str_pad('النطاق ' . (\$i + 1), 12) . ' | ';
}
echo str_pad('الإجمالي', 12) . PHP_EOL;

}

// عرض النطاقات
echo str_pad('النطاق', 30) . ' | ';
for (\$i = 0; \$i < 4; \$i++) {
    \$shortRange = substr(\$allStats[\$i]['range'], -5, 2) . '-' . substr(\$allStats[\$i]['range'], -2);
    echo str_pad(\$shortRange, 12) . ' | ';
}
echo PHP_EOL;

echo str_pad('عدد الفواتير', 30) . ' | ';
for (\$i = 0; \$i < 4; \$i++) {
    echo str_pad(\$allStats[\$i]['invoices'], 12) . ' | ';
}
echo PHP_EOL;

echo str_repeat('-', 30) . ' | ';
for (\$i = 0; \$i < 4; \$i++) {
    echo str_repeat('-', 12) . ' | ';
}
echo str_repeat('-', 12) . PHP_EOL;

// عرض المنتجات
foreach (array_keys(\$allProducts) as \$product) {
    echo str_pad(substr(\$product, 0, 28), 30) . ' | ';
    \$totalProduct = 0;
    for (\$i = 0; \$i < 4; \$i++) {
        \$qty = \$allStats[\$i]['products'][\$product] ?? 0;
        echo str_pad(number_format(\$qty), 12) . ' | ';
        \$totalProduct += \$qty;
    }
    echo str_pad(number_format(\$totalProduct), 12) . PHP_EOL;
}

echo str_repeat('-', 30) . ' | ';
for (\$i = 0; \$i < 4; \$i++) {
    echo str_repeat('-', 12) . ' | ';
}
echo str_repeat('-', 12) . PHP_EOL;

// الإجمالي
echo str_pad('الإجمالي الكلي', 30) . ' | ';
\$grandTotal = 0;
for (\$i = 0; \$i < 4; \$i++) {
    echo str_pad(number_format(\$allStats[\$i]['total']), 12) . ' | ';
    \$grandTotal += \$allStats[\$i]['total'];
}
echo str_pad(number_format(\$grandTotal), 12) . PHP_EOL;
