    النوع: decimal(12,2)
    القيمة الافتراضية: —

    النوع: enum('pending','approved','rejected','cancelled')
    القيمة الافتراضية: 'pending'

    النوع: bigint unsigned
    المفتاح: FOREIGN KEY
    يشير إلى: users.id

    القيمة الافتراضية: NULL

    النوع: text
    القيمة الافتراضية: NULL

    النوع: timestamp

  • updated_at
    النوع: timestamp
    القيمة الافتراضية: CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
